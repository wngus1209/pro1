      let i = 1;
      function prev() {
        i--;
        if (i < 1) i = 6;
        document.getElementsByTagName("img")[0].src =
          "images/photo" + 1 + ".jpg";
      }
      function next() {
        i++;
        if (i > 6) i = 1;
        let imgscr = document.getElementsByTagName("img")[0];
        let im = imgscr.getAttribute("src");
        imgscr.setAttribute("src", im.replace(im.slice(-5, -4), i));
      }
