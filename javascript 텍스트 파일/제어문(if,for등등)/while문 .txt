    <script>
      let text = "";
      let i = 0;
      /*while(조건) {

      while (i < 10) {
        text += i + "<br>";
        i++; //i = i +1 i += 1
      }
      document.write(text);
      let text2 = "";
      for (let i = 0; i < 10; i++) {
        text2 += i + ",";
      }
      document.write(text2);

      let text3 = "";
      let j = 9;
      do {
        text3 += "<br>The number is " + j;
        j--;
        if (j == 0) break;
      } while (j < 10);
      document.write(text3);
